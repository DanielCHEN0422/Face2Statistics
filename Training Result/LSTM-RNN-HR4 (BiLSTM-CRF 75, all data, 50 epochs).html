    include_top=False
)

BiLstm_layer = layers.Bidirectional(keras.layers.LSTM(50, return_sequences=True))(reshape)

merged_layer = layers.concatenate([dense1, dense2])

outt = layers.Dense(40,activation='softmax')(merged_layer)
model = keras.Model(inputs=img_input, outputs=outt)
model.compile(optimizer=keras.optimizers.Adam(),
             loss=keras.losses.CategoricalCrossentropy(),
             metrics=['accuracy'])
